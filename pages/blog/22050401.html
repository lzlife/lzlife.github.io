    1. 左侧文章列表

    默认只要在docs目录下的.md文件都会被收录为页面并展示，但是有些页面我不想展示出来，所以这里我在不想展示的页面的Front Matter里添加了一个 listShow: false 来控制页面要不要展示在列表里，这里需要修改一些主题文件

    1.首先需要修改列表这个组件（标签栏目用的也是这个组件），
路径是 docs\.vuepress\theme\components\NoteAbstract.vue

<template>
  <div class="abstract-wrapper">
    <NoteAbstractItem
      v-for="(item) in currentPageData"
      :key="item.path"
      :item="item"
      :currentPage="currentPage"
      :currentTag="currentTag"
    />
    <pagation
      class="pagation"
      :total="newData.length"  //这里需要使用筛选后的数据
      :currentPage="currentPage"
      @getCurrentPage="getCurrentPage"
    />
  </div>
</template>

<script>
import { defineComponent, ref, toRefs, computed, onMounted } from 'vue-demi'
import pagination from '@theme/mixins/pagination'
import NoteAbstractItem from './NoteAbstractItem'
import { useInstance } from '@theme/helpers/composable'

export default defineComponent({
  mixins: [pagination],
  components: { NoteAbstractItem },
  props: ['data', 'currentTag'],

  setup (props, ctx) {
    const instance = useInstance()

    const { data } = toRefs(props)

    const currentPage = ref(1)
    let newData=[]  //这里新增一个变量

    const currentPageData = computed(() => {
      const start = (currentPage.value - 1) * instance.$perPage
      const end = currentPage.value * instance.$perPage
      newData=data.value.filter((item)=>{
        return item.frontmatter.listShow !== false
      })  //对数据进行处理
      return newData.slice(start, end)
    })

    const getCurrentPage = (page) => {
      currentPage.value = page
      instance._setStoragePage(page)
      ctx.emit('paginationChange', page)
    }

    onMounted(() => {
      currentPage.value = instance._getStoragePage() || 1
    })

    return { currentPage, currentPageData, getCurrentPage,newData } //导出使用
  },
  watch: {
    $route () {
      this.currentPage = this._getStoragePage() || 1
    }
  }
})
</script>

路径是 docs\.vuepress\theme\components\PersonalInfo.vue

<template>
<div class="personal-info-wrapper">
  <img
    class="personal-img"
    v-if="$themeConfig.authorAvatar"
    :src="$withBase($themeConfig.authorAvatar)"
    alt="author-avatar"
  >
  <h3
    class="name"
    v-if="$themeConfig.author"
  >
    {{ $themeConfig.author }}
  </h3>
  <div class="num">
    <div>
      <h3>{{numList.length}}</h3>   //更改为numList.length
      <h6>{{$recoLocales.article}}</h6>
    </div>
    <div>
      <h3>{{$tags.list.length}}</h3>
      <h6>{{$recoLocales.tag}}</h6>
    </div>
  </div>
  <ul class="social-links">
    <li
      class="social-item"
      v-for="(item, index) in socialLinks"
      :key="index"
    >
      <reco-icon :icon="item.icon" :link="item.link" :style="{ color: item.color }" />
    </li>
  </ul>
  <hr>
</div>
</template>

<script>
import { defineComponent, computed ,toRefs} from 'vue-demi'  //加一个toRefs
import { RecoIcon } from '@vuepress-reco/core/lib/components'
import { getOneColor } from '@theme/helpers/other'
import { useInstance } from '@theme/helpers/composable'

export default defineComponent({
  components: { RecoIcon },
  props: ['listNum'], //接收父组件传过来的数据
  setup (props, ctx) {
    const instance = useInstance()

    const { listNum } = toRefs(props)  //这里再解构一下
    let numList=[]  //首先需要定义一个空的数组，页面初加载数据为空会报错
    if(listNum.value){  //判断有数据以后再执行筛选
      numList = listNum.value.filter((item,index)=>{
        return item.frontmatter.listShow !== false
      })
    }

    const socialLinks = computed(() => (
      instance.$themeConfig.blogConfig && instance.$themeConfig.blogConfig.socialLinks || []).map(item => {
      if (!item.color) item.color = getOneColor()
      return item
    }))

    return { socialLinks,numList } //导出使用
  }
})
</script>

    1. 深浅主题切换
    修改mode文件
路径是 docs\.vuepress\theme\components\Mode\index.vue

<template>
  <div class="color-btn" @click="selectMode">
    <i class="iconfontX icon-dark" v-if="!isDark"></i>
    <i class="iconfontX icon-light" v-else></i>
  </div>
</template>

<script>
import applyMode from './applyMode'

export default {
  name: 'UserSettings',
  data() {
    return {
      currentMode: 'light', //默认浅色
      isDark: false, //状态判断
    }
  },

  mounted() {
    // modePicker 开启时默认使用用户主动设置的模式
    this.currentMode =
      localStorage.getItem('mode') || this.$themeConfig.mode || 'light'
    if(this.currentMode!='light'){
      this.isDark=true
    }
    applyMode(this.currentMode)
  },

  methods: {
    selectMode() {
      if (this.currentMode == 'light') {
        this.currentMode = 'dark'
      } else {
        this.currentMode = 'light'
      }
      this.isDark = !this.isDark
      applyMode(this.currentMode)
      localStorage.setItem('mode', this.currentMode)
    },
  },
}
</script>
